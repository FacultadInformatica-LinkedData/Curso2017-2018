prefix vo:<http://example.org/myVocabulary#>

select ?a (sum(xsd:double(?c)) as ?Suma)  where 
{
?b vo:relatedDistrict ?a.
?b vo:numberOfAccidents ?c.
?b vo:hasAccidentType vo:BicycleFall
 }
ORDER BY DESC(?Suma)
